312 bool treetable_add(TreeTable *table, void *key, void *val)
313 {
314  RBNode *y = table->sentinel;
315  RBNode *x = table->root;
316 
317  int cmp;
318  while (x != table->sentinel) {
319  cmp = table->cmp(key, x->key);
320  y = x;
321 
322  if (cmp < 0) {
323  x = x->left;
324  } else if (cmp > 0) {
325  x = x->right;
326  } else {
327  x->value = val;
328  return true;
329  }
330  }
331  RBNode *n = table->mem_alloc(sizeof(RBNode));
332 
333  if (!n)
334  return false;
335 
336  n->value = val;
337  n->key = key;
338  n->parent = y;
339  n->left = table->sentinel;
340  n->right = table->sentinel;
341 
342  table->size++;
343 
344  if (y == table->sentinel) {
345  table->root = n;
346  n->color = RB_BLACK;
347  } else {
348  n->color = RB_RED;
349  if (table->cmp(key, y->key) < 0) {
350  y->left = n;
351  } else {
352  y->right = n;
353  }
354  rebalance_after_insert(table, n);
355  }
356  return true;
357 }

359 static void rebalance_after_insert(TreeTable *table, RBNode *z)
360 {
361  RBNode *y;
362 
363  while (z->parent->color == RB_RED) {
364  if (z->parent == z->parent->parent->left) {
365  y = z->parent->parent->right;
366  if (y->color == RB_RED) {
367  z->parent->color = RB_BLACK;
368  y->color = RB_BLACK;
369  z->parent->parent->color = RB_RED;
370  z = z->parent->parent;
371  } else {
372  if (z == z->parent->right) {
373  z = z->parent;
374  rotate_left(table, z);
375  }
376  z->parent->color = RB_BLACK;
377  z->parent->parent->color = RB_RED;
378  rotate_right(table, z->parent->parent);
379  }
380  } else {
381  y = z->parent->parent->left;
382  if (y->color == RB_RED) {
383  z->parent->color = RB_BLACK;
384  y->color = RB_BLACK;
385  z->parent->parent->color = RB_RED;
386  z = z->parent->parent;
387  } else {
388  if (z == z->parent->left) {
389  z = z->parent;
390  rotate_right(table, z);
391  }
392  z->parent->color = RB_BLACK;
393  z->parent->parent->color = RB_RED;
394  rotate_left(table, z->parent->parent);
395  }
396  }
397  }
398  table->root->color = RB_BLACK;
399 }

401 static void rebalance_after_delete(TreeTable *table, RBNode *x)
402 {
403  RBNode *w;
404 
405  while (x != table->root && x->color == RB_BLACK) {
406  if (x == x->parent->left) {
407  w = x->parent->right;
408  if (w->color == RB_RED) {
409  w->color = RB_BLACK;
410  x->parent->color = RB_RED;
411  rotate_left(table, x->parent);
412  w = x->parent->right;
413  }
414  if (w->left->color == RB_BLACK && w->right->color == RB_BLACK) {
415  w->color = RB_RED;
416  x = x->parent;
417  } else {
418  if (w->right->color == RB_BLACK) {
419  w->left->color = RB_BLACK;
420  w->color = RB_RED;
421  rotate_right(table, w);
422  w = x->parent->right;
423  }
424  w->color = x->parent->color;
425  x->parent->color = RB_BLACK;
426  w->right->color = RB_BLACK;
427  rotate_left(table, x->parent);
428  x = table->root;
429  }
430  } else {
431  w = x->parent->left;
432  if (w->color == RB_RED) {
433  w->color = RB_BLACK;
434  x->parent->color = RB_RED;
435  rotate_right(table, x->parent);
436  w = x->parent->left;
437  }
438  if (w->right->color == RB_BLACK && w->left->color == RB_BLACK) {
439  w->color = RB_RED;
440  x = x->parent;
441  } else {
442  if (w->left->color == RB_BLACK) {
443  w->right->color = RB_BLACK;
444  w->color = RB_RED;
445  rotate_left(table, w);
446  w = x->parent->left;
447  }
448  w->color = x->parent->color;
449  x->parent->color = RB_BLACK;
450  w->left->color = RB_BLACK;
451  rotate_right(table, x->parent);
452  x = table->root;
453  }
454  }
455  }
456  x->color = RB_BLACK;
457 }

495 static void remove_node(TreeTable *table, RBNode *z)
496 {
497  RBNode *x;
498  RBNode *y = z;
499 
500  int y_color = y->color;
501 
502  if (z->left == table->sentinel) {
503  x = z->right;
504  transplant(table, z, z->right);
505  } else if (z->right == table->sentinel) {
506  x = z->left;
507  transplant(table, z, z->left);
508  } else {
509  y = tree_min(table, z->right);
510  y_color = y->color;
511  x = y->right;
512  if (y->parent == z) {
513  x->parent = y;
514  } else {
515  transplant(table, y, y->right);
516  y->right = z->right;
517  y->right->parent = y;
518  }
519  transplant(table, z, y);
520  y->left = z->left;
521  y->left->parent = y;
522  y->color = z->color;
523  }
524  if (y_color == RB_BLACK)
525  rebalance_after_delete(table, x);
526 
527  table->mem_free(z);
528  table->size--;
529 }

583 {

586  x->left = y->right;

626 static RBNode *get_tree_node_by_key(TreeTable *table, void *key)
627 {
628  if (table->size == 0)
629  return NULL;
630 
631  RBNode *n = table->root;
632  RBNode *s = table->sentinel;
633 
634  int cmp;
635  do {
636  cmp = table->cmp(key, n->key);
637 
638  if (cmp < 0)
639  n = n->left;
640  else if (cmp > 0)
641  n = n->right;
642  else
643  return n;
644  } while (n != s && cmp != 0);
645 
646  return NULL;
647 }

649 static RBNode *get_successor_node(TreeTable *table, RBNode *x)
650 {
651  if (x == NULL)
652  return NULL;
653 
654  if (x->right != table->sentinel)
655  return tree_min(table, x->right);
656 
657  RBNode *y = x->parent;
658 
659  while (y != table->sentinel && x == y->right) {
660  x = y;
661  y = y->parent;
662  }
663  return y;
664 }

729 static int treetable_test(TreeTable *table, RBNode *node, int *nb)
730 {
731  if (node == table->sentinel) {
732  *nb = 1;
733  return RB_ERROR_OK;
734  }
735  /* check tree oreder */
736  if (node->left != table->sentinel) {
737  int cmp = table->cmp(node->left->key, node->key);
738  if (cmp >= 0)
739  return RB_ERROR_TREE_STRUCTURE;
740  }
741  if (node->right != table->sentinel) {
742  int cmp = table->cmp(node->right->key, node->key);
743  if (cmp <= 0)
744  return RB_ERROR_TREE_STRUCTURE;
745  }
746 
747  /* check red rule */
748  if (node->color == RB_RED && node->parent->color == RB_RED) {
749  return RB_ERROR_CONSECUTIVE_RED;
750  }
751 
752  int nb_left;
753  int nb_right;
754 
755  int left_err = treetable_test(table, node->left, &nb_left);
756 
757  /* propagate the descendant errors all the way up */
758  if (left_err != RB_ERROR_OK)
759  return left_err;
760 
761  int right_err = treetable_test(table, node->right, &nb_right);
762 
763  if (right_err != RB_ERROR_OK)

765 
766  /* check black rule */

768  return RB_ERROR_BLACK_HEIGHT;
769

772  else
